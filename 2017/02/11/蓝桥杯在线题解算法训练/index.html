6-1 递归求二项式系数值 

#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int getck(int k,int n){
    if(k==0||n==k){
            return 1;
    }else{
        return getck(k,n-1)+getck(k-1,n-1);
    }
}
int main()
{
    int k,n;
    cin>>k>>n;
    cout<<getck(k,n);           
    return 0;
}
